# 窗帘销售系统 · 数据库设计

> 基于 SQLAlchemy ORM 模型自动生成，数据类型兼容 SQLite
> 对应模型文件：`app/models.py`
> 版本：V2.2 | 更新：2026-04-14

---

## ER 关系图（简化）

```
orders ─────┬── customers（客户）
            ├── products（产品）── suppliers（供应商）── fabric_categories（布版）
            ├── employees（导购/安装工）
            └── installer_account（安装工账户）

orders ──── install_tasks（安装任务）── installer_account（安装工）
```

## 表设计

### 1. orders（订单表）

| 字段 | 类型 | 说明 |
|------|------|------|
| id | INTEGER PK | 主键 |
| order_no | VARCHAR(20) UNIQUE | 订单号，格式 YYMMDD+序号 |
| customer_id | INTEGER FK | 关联客户 |
| customer_name | VARCHAR(50) | 客户姓名（冗余） |
| customer_phone | VARCHAR(20) | 客户电话（冗余） |
| order_type | VARCHAR(20) | 窗帘/墙布/全屋 |
| status | VARCHAR(20) | 状态文字（显示用） |
| status_color | VARCHAR(20) | 状态颜色 |
| status_key | VARCHAR(30) | 状态key（V2.2八态） |
| quote_amount | DECIMAL(12,2) | 标价金额 |
| discount_amount | DECIMAL(12,2) | 优惠金额 |
| round_amount | DECIMAL(12,2) | 抹零金额 |
| amount | DECIMAL(12,2) | 应收金额 |
| received | DECIMAL(12,2) | 已收金额 |
| debt | DECIMAL(12,2) | 欠款 |
| order_date | VARCHAR(20) | 订单日期 |
| delivery_date | VARCHAR(20) | 交货日期 |
| delivery_method | VARCHAR(20) | 提货方式 |
| salesperson | VARCHAR(50) | 导购姓名 |
| history | JSON | 状态变更历史时间线 |
| items | JSON | 订单明细 JSON 数组 |
| content | VARCHAR(200) | 订单内容概述 |
| installer_id | INTEGER FK | 安装工ID（可空） |
| install_address | VARCHAR(300) | 安装地址 |
| install_date | DATE | 预约安装日期 |
| install_time_slot | VARCHAR(20) | 安装时段 |
| created_at | DATETIME | 创建时间 |

**索引**：`order_no`（唯一）、`customer_id`、`install_date`

## SQL 建表脚本

```sql
-- SQLite 语法（兼容 SQLAlchemy 自动生成）
-- 对应 app/models.py

CREATE TABLE orders (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_no VARCHAR(20) UNIQUE NOT NULL,
    customer_id INTEGER REFERENCES customers(id),
    customer_name VARCHAR(50),
    customer_phone VARCHAR(20),
    order_type VARCHAR(20),
    status VARCHAR(20) DEFAULT '待确认',
    status_color VARCHAR(20) DEFAULT 'pending',
    status_key VARCHAR(30) DEFAULT 'created',
    quote_amount DECIMAL(12,2) DEFAULT 0,
    discount_amount DECIMAL(12,2) DEFAULT 0,
    round_amount DECIMAL(12,2) DEFAULT 0,
    amount DECIMAL(12,2) DEFAULT 0,
    received DECIMAL(12,2) DEFAULT 0,
    debt DECIMAL(12,2) DEFAULT 0,
    order_date VARCHAR(20),
    delivery_date VARCHAR(20),
    delivery_method VARCHAR(20),
    salesperson VARCHAR(50),
    history JSON DEFAULT '[]',
    items JSON DEFAULT '[]',
    content VARCHAR(200),
    installer_id INTEGER REFERENCES installer_account(id),
    install_address VARCHAR(300),
    install_date DATE,
    install_time_slot VARCHAR(20),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_orders_order_no ON orders(order_no);
CREATE INDEX idx_orders_install_date ON orders(install_date);

CREATE TABLE customers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(50) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    type VARCHAR(20) DEFAULT '零售',
    address VARCHAR(300),
    community VARCHAR(100),
    source VARCHAR(50),
    salesperson VARCHAR(50),
    debt DECIMAL(12,2) DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE products (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(20) UNIQUE,
    name VARCHAR(100),
    supplier_id INTEGER REFERENCES suppliers(id),
    category_id INTEGER REFERENCES fabric_categories(id),
    product_type VARCHAR(20),
    classification VARCHAR(20),
    model VARCHAR(50),
    material VARCHAR(50),
    width INTEGER,
    weight INTEGER,
    unit_price DECIMAL(10,2),
    unit VARCHAR(10) DEFAULT '米',
    stock INTEGER DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE suppliers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(10),
    name VARCHAR(100),
    type VARCHAR(20),
    contact VARCHAR(50),
    phone VARCHAR(20),
    delivery_days INTEGER DEFAULT 7,
    address VARCHAR(300),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE fabric_categories (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(10),
    name VARCHAR(100),
    supplier_id INTEGER REFERENCES suppliers(id),
    description VARCHAR(200),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE employees (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(20),
    name VARCHAR(50),
    gender VARCHAR(10) DEFAULT '男',
    phone VARCHAR(20),
    position VARCHAR(20),
    department VARCHAR(20),
    max_discount DECIMAL(5,2) DEFAULT 1.0,
    round_limit INTEGER DEFAULT 0,
    is_installer BOOLEAN DEFAULT 0,
    status VARCHAR(20) DEFAULT '启用',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE installer_account (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(50) NOT NULL,
    phone VARCHAR(20) UNIQUE NOT NULL,
    password_hash VARCHAR(200),
    wechat VARCHAR(100),
    status VARCHAR(20) DEFAULT 'active',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE install_task (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_id INTEGER NOT NULL REFERENCES orders(id),
    order_no VARCHAR(20),
    installer_id INTEGER REFERENCES installer_account(id),
    install_date DATE NOT NULL,
    install_time_slot VARCHAR(20),
    address VARCHAR(300) NOT NULL,
    customer_name VARCHAR(50) NOT NULL,
    customer_phone VARCHAR(20) NOT NULL,
    raw_customer_phone VARCHAR(20),
    order_content VARCHAR(500),
    priority VARCHAR(20) DEFAULT 'normal',
    status VARCHAR(20) DEFAULT 'pending',
    completed_at DATETIME,
    completion_remark VARCHAR(1000),
    navigate_url VARCHAR(500),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_task_install_date ON install_task(install_date);
CREATE INDEX idx_task_status ON install_task(status);

CREATE TABLE purchase (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    po_no VARCHAR(30) UNIQUE,
    supplier_id INTEGER,
    supplier_name VARCHAR(100),
    contact VARCHAR(50),
    phone VARCHAR(30),
    amount DECIMAL(12,2) DEFAULT 0,
    paid DECIMAL(12,2) DEFAULT 0,
    debt DECIMAL(12,2) DEFAULT 0,
    status VARCHAR(30) DEFAULT '待采购',
    order_date DATE,
    expected_date DATE,
    items JSON DEFAULT '[]',
    remark VARCHAR(500),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE warehouse_records (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    record_type VARCHAR(10),
    product_id INTEGER,
    product_name VARCHAR(200),
    qty DECIMAL(10,2),
    unit VARCHAR(10) DEFAULT '米',
    remark VARCHAR(300),
    operator VARCHAR(50),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE finance_records (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    record_type VARCHAR(20),
    order_id INTEGER,
    order_no VARCHAR(30),
    customer_name VARCHAR(100),
    amount DECIMAL(12,2),
    method VARCHAR(30) DEFAULT '转账',
    operator VARCHAR(50),
    remark VARCHAR(300),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE store_config (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    store_name VARCHAR(100) DEFAULT '金典软装',
    store_phone VARCHAR(30),
    store_address VARCHAR(300),
    workday_start VARCHAR(10) DEFAULT '09:00',
    workday_end VARCHAR(10) DEFAULT '18:00',
    latitude DECIMAL(10,6),
    longitude DECIMAL(10,6),
    qrcode_base_url VARCHAR(500),
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE sms_code (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    phone VARCHAR(20),
    code VARCHAR(10),
    code_type VARCHAR(30) DEFAULT 'installer_login',
    expires_at DATETIME NOT NULL,
    used BOOLEAN DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_sms_phone ON sms_code(phone);

CREATE TABLE customer_progress_query_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_no VARCHAR(20),
    phone_hint VARCHAR(10) NOT NULL,
    query_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    ip_address VARCHAR(50),
    user_agent VARCHAR(500)
);

CREATE INDEX idx_query_order_no ON customer_progress_query_log(order_no);
```
